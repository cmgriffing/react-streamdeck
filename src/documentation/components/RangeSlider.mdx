# RangeSlider

import { Props, Playground } from "docz";
import { SDRangeSlider, SDRangeWithLabels } from "../../components/RangeSlider";
import DemoStateContainer from "react-component-component";

## SDRangeSlider (Base)

This is the base List component. Other List components wrap this one with other default config options.

### Example

<Playground>
  <div class="docz-sdpi-wrapper">
    <DemoStateContainer
      initialState={{
        rangeSliderState: ''
      }}
    >
      {({setState, state: { rangeSliderState }}) => (
        <SDRangeSlider
          label="Range Slider"
          min={0}
          max={100}
          value={+rangeSliderState}
          onChange={event => {
            setState({
              rangeSliderState: +event.target.value
            });
          }}
        />
      )}
    </DemoStateContainer>
  </div>
</Playground>

<Props of={SDRangeSlider} />
